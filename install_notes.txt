# CREATE CPU VENV 
rm -rf venvs requirements_CPU.txt

python3 -m venv venvs/CPU_venv
source venvs/CPU_venv/bin/activate

venvs/CPU_venv/bin/python3 -m pip install --upgrade pip
venvs/CPU_venv/bin/python3 -m pip install torch==1.11.0+cpu torchvision==0.12.0+cpu torchaudio==0.11.0 --extra-index-url https://download.pytorch.org/whl/cpu
venvs/CPU_venv/bin/python3 -m pip install torch-scatter torch-sparse torch-cluster torch-spline-conv torch-geometric -f https://data.pyg.org/whl/torch-1.11.0+cpu.html
venvs/CPU_venv/bin/python3 -m pip install ogb einops

pip freeze >> ./requirements_CPU.txt
deactivate 

# TEST VENV
python3 -m venv venvs/TEST
source venvs/TEST2/bin/activate
venvs/TEST2/bin/python3 -m pip install -r ./requirements_GPU.txt
